define Gyro turn right power left power degree
set yaw angle to 0
start moving at right power left power % speed
wait until ((abs of yaw angle) > degree)
stop moving

define Gyro turn right power left power degrees >/</=
start moving at right power left power % speed
if (>/</= = >) then
   wait until (yaw angle > degrees)
if (>/</= = <) then
   wait until (yaw angle < degrees)
if (>/</= = =) then
   wait until (yaw angle = degrees)
stop moving
