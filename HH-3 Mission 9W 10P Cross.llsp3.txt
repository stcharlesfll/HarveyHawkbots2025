define Gryo Turn Left Power Right Power Degree
if (Left Power > Right Power) then
   start moving at Left Power Right Power % speed
   wait until ((abs of yaw angle) > Degree)
else
   start moving at Left Power Right Power % speed
   wait until ((abs of yaw angle) < Degree)
stop moving
