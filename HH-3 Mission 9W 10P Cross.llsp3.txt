define Gryo Turn Left Power Right Power Degree
if (Left Power > Right Power) then
   start moving at Left Power Right Power % speed
   wait until ((abs of yaw angle) > Degree)
else
   start moving at Left Power Right Power % speed
   wait until ((abs of yaw angle) < Degree)
stop moving

define Gryo Turn No Reset Left Wheel Speed Right Wheel Speed Degree Greater/Less
start moving at Left Wheel Speed Right Wheel Speed % speed
if (Greater/Less = >) then
   wait until (yaw angle > Degree)
if (Greater/Less = <) then
   wait until (yaw angle < Degree)
if (Greater/Less = =) then
   wait until (yaw angle = Degree)
stop moving
